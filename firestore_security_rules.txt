rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection:
    // - Authenticated users can read any user profile (useful for admin/provider lists).
    // - Authenticated users can create their own user profile document (during registration).
    // - Authenticated users can update their own profile document.
    // - Deleting user profiles from the client is not allowed.
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.resource.data.uid == request.auth.uid;
      allow update: if request.auth != null && request.auth.uid == userId;
      allow delete: if false;
    }

    // Bookings collection:
    // - Only authenticated users can read/write bookings.
    match /bookings/{bookingId} {
      allow read, write: if request.auth != null;
    }

    // Deny all other read/write operations by default for any other collection
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
